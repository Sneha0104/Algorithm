Max Heap

heapify(int arr[], int size, int i) {

largest = i,left = 2*i + 1,
right = 2*i + 2

if (left < size && arr[left]>arr[largest])
largest = left

if (right < size && arr[right]> arr[largest])
largest = right

if (largest != i) {
temp = arr[i];
arr[i]= arr[largest];
arr[largest] = temp;
heapify(arr, size, largest); } }

void heapSort(int arr[], int size) {

for (i = size / 2 - 1 to 0)
heapify(arr, size, i);

for (i=size-1 to 0) {
swap(arr[0],arr[i])
heapify(arr, i, 0)
} }
