bool push(int x) {

    cout << "Stack Overflow";
    return false;
}
  else {
    a[++top] = x;
   cout << x << " pushed into stack\n";
   return true;
}

}

int pop() {

  if (top < 0) {
   cout << "Stack Underflow";
   return 0;
}
 else {
   int x = a[top--];
   return x;
}

}

Time Complexity

Push: Insert element to top of stack
Time Complexity: O(1)

Pop: Remove element from the top of stack
Time Complexity: O(1)
